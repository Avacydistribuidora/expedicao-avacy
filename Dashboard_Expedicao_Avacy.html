🔗 Configurar Sincronização — Supabase

Configure um banco Supabase gratuito para sincronizar dados entre múltiplos usuários em tempo real. Todos verão os mesmos pedidos e status.

1. Crie uma conta gratuita em supabase.com e crie um projeto
2. No SQL Editor do Supabase, execute este script:
DROP TABLE IF EXISTS avacy_status;
DROP TABLE IF EXISTS avacy_records;
CREATE TABLE avacy_records (
  nota TEXT PRIMARY KEY, fat TEXT, loja TEXT,
  nome_cliente TEXT, vendedor TEXT, transportadora TEXT,
  data_emissao TEXT, romaneio TEXT, mensagem TEXT,
  origem TEXT, xconf TEXT,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE avacy_status (
  nota TEXT PRIMARY KEY,
  status TEXT DEFAULT 'Faturado', obs TEXT DEFAULT '',
  priority BOOLEAN DEFAULT FALSE,
  ts_fat TEXT, ts_sep TEXT, ts_sepd TEXT,
  ts_conf TEXT, ts_exp TEXT, ts_pend TEXT, pend_prev TEXT,
  scan_log JSONB, sep_log JSONB, pend_log JSONB,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS avacy_users (
  username TEXT PRIMARY KEY,
  password TEXT NOT NULL,
  role TEXT DEFAULT 'user',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE avacy_records ENABLE ROW LEVEL SECURITY;
ALTER TABLE avacy_status  ENABLE ROW LEVEL SECURITY;
ALTER TABLE avacy_users   ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON avacy_records FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all" ON avacy_status  FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all" ON avacy_users   FOR ALL USING (true) WITH CHECK (true);
-- Colunas adicionais (execute se já tiver o schema criado):
ALTER TABLE avacy_records ADD COLUMN IF NOT EXISTS volume TEXT;
ALTER TABLE avacy_records ADD COLUMN IF NOT EXISTS chave_nfe TEXT;
